//递归算法，依次输出某整数

#include<stdio.h>

// 递归算法，依次输出某整数的每一位数字
void frecursion(int n)
{
// 递归终止条件：当 n 为 0 时，停止递归    
    if (n == 0)
        return;
    
    frecursion(n / 10);
    printf("%d\n", n % 10);
}

int main()
{
    frecursion(321);
    return 0;
}
